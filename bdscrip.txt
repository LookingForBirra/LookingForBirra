CREATE TABLE amigos (
  id_amigos INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  usuario_id_usuario INTEGER UNSIGNED NOT NULL,
  amigo_id_usuario INTEGER UNSIGNED NOT NULL,
  PRIMARY KEY(id_amigos),
  INDEX amigos_FKIndex1(usuario_id_usuario)
);

CREATE TABLE BAR (
  id_BAR INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Nombre VARCHAR NULL,
  Direccion VARCHAR(255) NULL,
  Tipo INTEGER UNSIGNED NULL,
  Aforo INTEGER UNSIGNED NULL,
  Link RRSS INTEGER UNSIGNED NULL,
  Tematica VARCHAR(255) NULL,
  Imagen VARCHAR(255) NULL,
  Zona VARCHAR(255) NULL,
  PRIMARY KEY(id_BAR)
);

CREATE TABLE BAR_has_CERVEZA (
  BAR_id_BAR INTEGER UNSIGNED NOT NULL,
  CERVEZA_id_CERVEZA INTEGER UNSIGNED NOT NULL,
  Precio FLOAT NULL,
  Disponibilidad_de_cervezas INTEGER UNSIGNED NULL,
  PRIMARY KEY(BAR_id_BAR, CERVEZA_id_CERVEZA),
  INDEX BAR_has_CERVEZA_FKIndex1(BAR_id_BAR),
  INDEX BAR_has_CERVEZA_FKIndex2(CERVEZA_id_CERVEZA)
);

CREATE TABLE CERVEZA (
  id_CERVEZA INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  Nombre VARCHAR(255) NULL,
  Tipo INTEGER UNSIGNED NULL,
  Graduacion FLOAT NULL,
  Procedencia VARCHAR(255) NULL,
  Envase INTEGER UNSIGNED NULL,
  Imagen VARCHAR(255) NULL,
  PRIMARY KEY(id_CERVEZA)
);

CREATE TABLE preferencias (
  id_preferencias INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  usuario_id_usuario INTEGER UNSIGNED NOT NULL,
  CERVEZA_id_CERVEZA INTEGER UNSIGNED NOT NULL,
  BAR_id_BAR INTEGER UNSIGNED NOT NULL,
  PRIMARY KEY(id_preferencias),
  INDEX preferencias_FKIndex2(BAR_id_BAR),
  INDEX preferencias_FKIndex3(CERVEZA_id_CERVEZA),
  INDEX preferencias_FKIndex3(usuario_id_usuario)
);

CREATE TABLE usuario (
  id_usuario INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  BAR_id_BAR INTEGER UNSIGNED NOT NULL,
  Nombre VARCHAR(100) NULL,
  Apellido1 VARCHAR(100) NULL,
  Apellido2 VARCHAR(100) NULL,
  Edad INTEGER UNSIGNED NULL,
  Nivel VARCHAR(100) NULL,
  Puntos INTEGER UNSIGNED NULL,
  Ciudad VARCHAR(100) NULL,
  Sexo VARCHAR(100) NULL,
  Telefono INTEGER UNSIGNED NULL,
  Localidad VARCHAR(100) NULL,
  Nick VARCHAR(100) NULL,
  Pass VARCHAR(100) NULL,
  Correo-e VARCHAR(100) NULL,
  Foto VARCHAR(100) NULL,
  PRIMARY KEY(id_usuario),
  INDEX usuario_FKIndex1(BAR_id_BAR)
);
